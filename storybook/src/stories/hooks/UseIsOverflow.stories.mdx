# useIsOverflow

Use this function to detect if an HTML element is overflowing.

* Return Value: true or false if the element is overflowing.

* Function Signature: 
``` ts dark
function useIsOverflow<RefType extends HTMLElement = HTMLElement>(ref: React.MutableRefObject<RefType>, isVerticalOverflow?: boolean, callback?: (hasOverflow: boolean) => void): boolean
```

* Parameters:

<TypesTable rows={[
    {
        Id: 1,
        name: 'ref',
        type: 'React.MutableRefObject<RefType>',
        description: 'A `React.MutableRefObject<RefType>`, which means that `RefType` should extend `HTMLElement`.',
        required: true
    },
    {
        Id: 2,
        name: 'isVerticalOverflow',
        type: 'boolean',
        description: 'If true, the function will check if the element is overflowing vertically.',
        required: false
    },
    {
        Id: 3,
        name: 'callback',
        type: '(hasOverflow: boolean) => void',
        description: 'A callback function that will be called with the result of the overflow check.',
        required: false
    }
]}/>

* Example:
``` tsx dark
import {useIsOverflow} from 'trentim-react-sdk';
import {useRef} from 'react';

function SomeComponent() {
    const ref = useRef<HTMLDivElement>(null);
    const isOverflowing = useIsOverflow(ref, false, (hasOverflow) => {
        console.log("Element is overflowing: " + hasOverflow);
    });
    //Than do something if isOverflowing is true...

    return (<div ref={ref}>...<div>);
}
```
